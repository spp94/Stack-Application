import java.util.*; 
public class Expression {
	public boolean isBalanced(String input){
		Stack<Character> stack = new Stack<Character>();
		for(char ch : input.toCharArray()){
			if(ch == '('){
				stack.push(ch);
			}
			else if(ch == ')'){
				if(stack.isEmpty()){
					return false;
				}
				stack.pop();
			}
			else{
				continue;
			}
		}
		return stack.isEmpty();
	}
}
